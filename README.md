## Install

```bash
npm install --save code-markup-parser
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'code-markup-parser'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
